```python
#This function prints the grid of Minesweeper Game as the game progresses
def print_grid():
    print('-' * (5 * M + M + 1))
    for i in range(N):
        print(end='|')
        for j in range(M):
            if viewed_grid[i][j] == 0:
                e = '  .  '
            elif viewed_grid[i][j] == 2:
                e = '  F  '
            elif grid[i][j] == 0:
                e = '     '
            elif grid[i][j] == -1:
                e = '  *  '
            else:
                e = '  ' + str(grid[i][j]) + '  '
            print(e, end='|')
        print()
        print('-' * (5 * M + M + 1))
```
